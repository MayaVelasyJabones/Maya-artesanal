function calcularCiclo() { const inputFecha = document.getElementById('fechaNacimiento').value; if (!inputFecha) { alert("Por favor, ingresa una fecha válida."); return; } // Obtenemos todos los dígitos de la fecha (ej: 1995-05-15 -> 19950515) let digitos = inputFecha.replace(/-/g, ''); let suma = 0; // Primera suma de todos los números for (let d of digitos) { suma += parseInt(d); } // Reducción a un solo dígito (mientras sea mayor a 9, se vuelve a sumar) while (suma > 9) { suma = suma.toString().split('').reduce((acc, curr) => acc + parseInt(curr), 0); } // Mostrar resultados y asignar mensajes const caja = document.getElementById('resultadoCiclo'); const spanNumero = document.getElementById('numeroFinal'); const pMensaje = document.getElementById('mensajeCiclo'); const hNombre = document.getElementById('nombreCiclo'); caja.style.display = 'block'; spanNumero.innerText = suma; // Contenido basado en tus ciclos const guiaCiclos = { 1: { nombre: "Ciclo de Nuevos Comienzos", texto: "Es momento de la independencia y de sembrar proyectos con fuerza propia." }, 2: { nombre: "Ciclo de Dualidad y Empatía", texto: "Tiempo de trabajar la paciencia, las relaciones y el equilibrio emocional." }, 3: { nombre: "Ciclo de Expansión Creativa", texto: "La comunicación y la alegría fluyen. Expresa tu verdad sin miedo." }, 4: { nombre: "Ciclo de Estructura", texto: "Enfoque en el orden, el trabajo constante y la construcción de bases sólidas." }, 5: { nombre: "Ciclo de Libertad", texto: "Cambios inesperados y aventuras. Fluye con el movimiento de la vida." }, 6: { nombre: "Ciclo de Armonía Social", texto: "Ideal para el cuidado del hogar, la familia y la responsabilidad amorosa." }, 7: { nombre: "Ciclo de Sabiduría", texto: "Un tiempo de introspección, estudio y conexión espiritual profunda." }, 8: { nombre: "Ciclo de Manifestación", texto: "Poder personal y resultados materiales. Cosecha lo que has trabajado." }, 9: { nombre: "Ciclo de Cierre", texto: "Culminación y perdón. Limpia tu espacio para recibir el próximo gran ciclo." } }; hNombre.innerText = guiaCiclos[suma].nombre; pMensaje.innerText = guiaCiclos[suma].texto; // Desplazar la pantalla suavemente hacia el resultado caja.scrollIntoView({ behavior: 'smooth' }); }
